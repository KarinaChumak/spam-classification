Text preprocessing includes next steps:

  • deleting all unicode characters
  • substituting all email addresses with emailaddr
  • substituting all url with httpaddr, saving most common top-level-domains('org','com','me','ly','cab')
  • substituting all money symbols with moneysymb
  • substituting all phone numbers with phonenumbr
  • substituting all numbers with numbr
  • remove all punctuation and extra whitespaces
In [41]:
def preprocess_text(messy_string):
    cleaned = re.compile(r'[\u064B-\u0652\u06D4\u0670\u0674\u06D5-\u06ED]+',re.UNICODE).sub('', messy_string)
    cleaned = re.sub(r'\b[\w\-.]+?@\w+?\.\w{2,4}\b', 'emailaddr', cleaned )
    if has_url(cleaned):
        suffix = get_suffix(cleaned)
        if suffix in ['org','com','me','ly','cab']:
            cleaned += ' {}'.format(suffix)
    cleaned = re.sub(r'http[s]?://(?:[a-zA-Z]|[0-9]|[$-_@.&+]|[!*\(\),]|(?:%[0-9a-fA-F][0-9a-fA-F]))+', 'httpaddr',
                     cleaned)
    cleaned = re.sub(r'£|\$', 'moneysymb', cleaned)
    cleaned = re.sub(
        r'\b(\+\d{1,2}\s)?\d?[\-(.]?\d{3}\)?[\s.-]?\d{3}[\s.-]?\d{4}\b',
        'phonenumbr', cleaned)
    cleaned = re.sub(r'\d+(\.\d+)?', 'numbr ', cleaned)
    cleaned = re.sub(r'[^\w\d\s]', ' ', cleaned)
    cleaned = re.sub(r'\s+', ' ', cleaned)
    cleaned = re.sub(r'^\s+|\s+?$', '', cleaned.lower())
    return ' '.join(
        porter.stem(term) 
        for term in cleaned.split()
        if term not in set(stop_words)
    )
